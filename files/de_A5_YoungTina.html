def process_parquet_file(input_file, output_file):
  #read the parquet file into a DataFrame
  parDF = pd.read_parquet(input_file)

  #Organizing the rank of customer's orders by price (highest rank = 1)
  parDF['order_rank'] = parDF.groupby('customerid')['price'].rank(ascending=False)

  #write it back to a dataframe as a new name
  parDF.to_parquet(output_file, index=False)

def read_parquet_file(parquet_file_path):
  """
  Read a Parquet file into a DataFrame

  :param parquet_file_path: Path to the Parquet file
  :return: DataFrame containing the data from the Parquet file
  """
  df_parquet = pd.read_parquet(parquet_file_path)
  print(f"'{parquet_file_path}' data read successful")
  return df_parquet

def create_table_task4(conn, table_name, df_parquet):
  """
  Create new table in SQLite database

  :param conn: SQLite connection object
  :param table_name: Name of the table to be created
  :param df_parquet: DataFrame containing the data to be used for the table
  """
  df_parquet.to_sql(table_name, conn, if_exists="replace", index=False)
  print(f"Table '{table_name}' created successfully")

def extract_data():
    """
    Extracts json data from url and save it locally.
    """
    url = 'https://raw.githubusercontent.com/matthewpecsok/data_engineering/main/data/customer_orders.json'
    response = requests.get(url)

    if response.status_code == 200:
      data = response.json()
      with open("data.json", "w") as file:
        json.dump(data, file, indent=4)
      print("Data successfully extracted and saved to 'data.json'")
    else:
      print(f"Failed to retrieve data. Status code: {response.status_code}")

def process_data():
    """
    Process the json file into parquet file.
    """
    df_lab5 = pd.read_json("data.json")
    print("Data successfully read into DataFrame")

    df_final = None

    if 'orders' in df_lab5.columns:
      df_exploded = df_lab5.explode("orders").reset_index(drop=True)
      orders_normalized = pd.json_normalize(df_exploded["orders"])
      df_final = pd.concat([df_exploded.drop(columns=["orders"]), orders_normalized], axis=1)

      df_final.to_parquet("data.parquet", index=False)
      print("Data successfully processed and saved as data.parquet")
    else:
      print("'orders' column is not found in the DataFrame.")

    return df_final

def process_parquet_file():
    """
    Processes the Parquet file to add ranking and saves the result.
    """
    input_file = "data.parquet"  # Default input file path
    output_file = "data_rank.parquet"  # Default output file path

    # Read the parquet file into a DataFrame
    parDF = pd.read_parquet(input_file)

    # Organize the rank of customer's orders by price (highest rank = 1)
    parDF['order_rank'] = parDF.groupby('customerid')['price'].rank(ascending=False)

    # Write it back to a dataframe as a new name
    parDF.to_parquet(output_file, index=False)
    print(f"Processed Parquet file saved as '{output_file}'")
