In [71]:
cur_cust_dest.execute("""
  insert into final_table (customerid, age, first_name, last_name, created_date, email_type, email)
  select
  c.customerid,
  abs(c.age) as age, --fixed negative age issues
  c.first_name,
  c.last_name,
  c.created_date,
  CASE
    when ce.email_type is null then 'Missing'
    else ce.email_type
  end as email_type,
  case
    when ce.email is null then 'Missing'
    else ce.email
  end as email
  from task3_customer_data c
  left join task3_customer_emails ce on c.customerid = ce.customerid
  where c.created_date >= '2022-01-01';
  """)

con_cust_dest.commit()
